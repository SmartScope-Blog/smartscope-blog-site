    # socratic_learning_system.py
import openai
import json
from typing import Dict, List, Optional
from dataclasses import dataclass
from enum import Enum

class LearningPhase(Enum):
    INITIAL_ASSESSMENT = "initial_assessment"
    GUIDED_DISCOVERY = "guided_discovery"
    CONCEPT_REINFORCEMENT = "concept_reinforcement"
    SELF_REFLECTION = "self_reflection"
    MASTERY_CHECK = "mastery_check"

@dataclass
class LearningSession:
    user_id: str
    topic: str
    current_phase: LearningPhase
    understanding_level: int  # 1-10
    conversation_history: List[Dict]
    learning_objectives: List[str]
    misconceptions_detected: List[str]

class SocraticLearningEngine:
    def __init__(self, openai_api_key: str):
        self.client = openai.OpenAI(api_key=openai_api_key)
        self.session_store = {}

    def create_session(self, user_id: str, topic: str) -> LearningSession:
        """新しい学習セッションを開始"""
        session = LearningSession(
            user_id=user_id,
            topic=topic,
            current_phase=LearningPhase.INITIAL_ASSESSMENT,
            understanding_level=5,
            conversation_history=[],
            learning_objectives=self._generate_objectives(topic),
            misconceptions_detected=[]
        )
        self.session_store[user_id] = session
        return session

    def _generate_objectives(self, topic: str) -> List[str]:
        """トピックに基づく学習目標生成"""
        response = self.client.chat.completions.create(
            model="gpt-4o",
            messages=[
                {"role": "system", "content": """
あなたは学習目標設定の専門家です。
与えられたトピックに対して、段階的で具体的な学習目標を3-5個生成してください。
各目標は測定可能で達成可能なものにしてください。
"""},
                {"role": "user", "content": f"トピック: {topic}"}
            ],
            temperature=0.3
        )

        # レスポンスを解析して目標リストを抽出
        objectives_text = response.choices[0].message.content
        return [obj.strip() for obj in objectives_text.split('\n') if obj.strip()]

    def generate_socratic_question(self, session: LearningSession, 
                                 student_response: str = None) -> str:
        """ソクラテス式質問生成"""
        system_prompt = f"""
あなたはソクラテス式教育法の専門家です。
学習者の理解を深めるために、直接的な答えを避け、
思考を促進する質問を生成してください。

現在のフェーズ: {session.current_phase.value}
トピック: {session.topic}
理解レベル: {session.understanding_level}/10
学習目標: {', '.join(session.learning_objectives)}
検出された誤解: {', '.join(session.misconceptions_detected)}

ガイドライン:
1. 直接的な答えは提供しない
2. 段階的な思考プロセスを促す
3. 具体例や類推を使用
4. 学習者の既存知識に関連付け
5. メタ認知を促進する質問も含める
"""

        messages = [{"role": "system", "content": system_prompt}]

        # 会話履歴を追加
        for msg in session.conversation_history[-5:]:  # 直近5件のみ
            messages.append(msg)

        if student_response:
            messages.append({"role": "user", "content": student_response})

        response = self.client.chat.completions.create(
            model="gpt-4o",
            messages=messages,
            temperature=0.7,
            max_tokens=300
        )

        return response.choices[0].message.content

    def analyze_understanding(self, session: LearningSession, 
                            student_response: str) -> Dict:
        """学習者の理解度分析"""
        analysis_prompt = f"""
以下の学習者の回答を分析し、理解度と誤解を特定してください。

トピック: {session.topic}
学習目標: {', '.join(session.learning_objectives)}
学習者の回答: {student_response}

以下の形式でJSON回答してください:
{{
    "understanding_level": 1-10の数値,
    "misconceptions": ["誤解1", "誤解2"],
    "strengths": ["理解できている点1", "理解できている点2"],
    "next_focus_areas": ["次に重点的に学ぶべき領域1", "領域2"],
    "phase_recommendation": "適切な次のフェーズ"
}}
"""

        response = self.client.chat.completions.create(
            model="gpt-4o",
            messages=[
                {"role": "system", "content": "あなたは学習分析の専門家です。"},
                {"role": "user", "content": analysis_prompt}
            ],
            temperature=0.2
        )

        try:
            analysis = json.loads(response.choices[0].message.content)
            # セッション更新
            session.understanding_level = analysis["understanding_level"]
            session.misconceptions_detected.extend(analysis["misconceptions"])
            return analysis
        except json.JSONDecodeError:
            return {"error": "分析結果の解析に失敗しました"}

    def conduct_learning_session(self, user_id: str, 
                                student_input: str) -> Dict:
        """学習セッション実行"""
        session = self.session_store.get(user_id)
        if not session:
            return {"error": "セッションが見つかりません"}

        # 学習者の回答を分析
        if student_input:
            analysis = self.analyze_understanding(session, student_input)
            session.conversation_history.append({
                "role": "user", 
                "content": student_input
            })

        # 次の質問を生成
        next_question = self.generate_socratic_question(session, student_input)
        session.conversation_history.append({
            "role": "assistant", 
            "content": next_question
        })

        # フェーズ進行の判定
        self._update_learning_phase(session)

        return {
            "question": next_question,
            "understanding_level": session.understanding_level,
            "current_phase": session.current_phase.value,
            "progress": len(session.conversation_history) / 20 * 100  # 概算進捗
        }

    def _update_learning_phase(self, session: LearningSession):
        """学習フェーズの更新ロジック"""
        conversation_count = len(session.conversation_history)
        understanding = session.understanding_level

        if conversation_count < 4:
            session.current_phase = LearningPhase.INITIAL_ASSESSMENT
        elif understanding < 6:
            session.current_phase = LearningPhase.GUIDED_DISCOVERY
        elif understanding < 8:
            session.current_phase = LearningPhase.CONCEPT_REINFORCEMENT
        elif understanding < 9:
            session.current_phase = LearningPhase.SELF_REFLECTION
        else:
            session.current_phase = LearningPhase.MASTERY_CHECK

# 使用例
if __name__ == "__main__":
    engine = SocraticLearningEngine("your-openai-api-key")

    # セッション開始
    session = engine.create_session("user123", "機械学習の基礎")

    # 学習セッション実行
    result = engine.conduct_learning_session("user123", "機械学習について教えて")
    print(f"質問: {result['question']}")
    print(f"理解レベル: {result['understanding_level']}/10")
    print(f"現在のフェーズ: {result['current_phase']}")

    1.2 Web インターフェース実装¶

    # app.py - Flask Web アプリケーション
from flask import Flask, render_template, request, jsonify, session
from socratic_learning_system import SocraticLearningEngine
import os

app = Flask(__name__)
app.secret_key = os.environ.get('SECRET_KEY', 'development-key')

# 学習エンジン初期化
learning_engine = SocraticLearningEngine(
    openai_api_key=os.environ.get('OPENAI_API_KEY')
)

@app.route('/')
def index():
    return render_template('learning_interface.html')

@app.route('/start_session', methods=['POST'])
def start_session():
    data = request.json
    topic = data.get('topic')
    user_id = session.get('user_id', 'anonymous')

    # セッション開始
    learning_session = learning_engine.create_session(user_id, topic)

    # 最初の質問生成
    result = learning_engine.conduct_learning_session(user_id, None)

    return jsonify(result)

@app.route('/submit_response', methods=['POST'])
def submit_response():
    data = request.json
    user_response = data.get('response')
    user_id = session.get('user_id', 'anonymous')

    result = learning_engine.conduct_learning_session(user_id, user_response)

    return jsonify(result)

@app.route('/get_progress', methods=['GET'])
def get_progress():
    user_id = session.get('user_id', 'anonymous')
    session_data = learning_engine.session_store.get(user_id)

    if not session_data:
        return jsonify({"error": "セッションが見つかりません"})

    return jsonify({
        "understanding_level": session_data.understanding_level,
        "current_phase": session_data.current_phase.value,
        "objectives_completed": len([
            obj for obj in session_data.learning_objectives 
            if session_data.understanding_level >= 8
        ]),
        "total_objectives": len(session_data.learning_objectives),
        "conversation_count": len(session_data.conversation_history)
    })

if __name__ == '__main__':
    app.run(debug=True)

    <script>
        let sessionActive = false;

        function startSession() {
            const topic = document.getElementById('topic-input').value;
            if (!topic.trim()) {
                alert('学習トピックを入力してください');
                return;
            }

            fetch('/start_session', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ topic: topic })
            })
            .then(response => response.json())
            .then(data => {
                if (data.error) {
                    alert('エラー: ' + data.error);
                    return;
                }

                document.getElementById('session-setup').style.display = 'none';
                document.getElementById('learning-interface').style.display = 'block';

                addMessage(data.question, 'ai');
                updateProgress(data.understanding_level, data.current_phase, data.progress);
                sessionActive = true;
            })
            .catch(error => {
                console.error('Error:', error);
                alert('セッション開始に失敗しました');
            });
        }

        function submitResponse() {
            const userInput = document.getElementById('user-input');
            const response = userInput.value.trim();

            if (!response || !sessionActive) return;

            addMessage(response, 'user');
            userInput.value = '';

            fetch('/submit_response', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ response: response })
            })
            .then(response => response.json())
            .then(data => {
                if (data.error) {
                    alert('エラー: ' + data.error);
                    return;
                }

                addMessage(data.question, 'ai');
                updateProgress(data.understanding_level, data.current_phase, data.progress);
            })
            .catch(error => {
                console.error('Error:', error);
                alert('回答の送信に失敗しました');
            });
        }

        function addMessage(message, sender) {
            const chatContainer = document.getElementById('chat-container');
            const messageDiv = document.createElement('div');
            messageDiv.className = `message ${sender}-message`;
            messageDiv.innerHTML = `<strong>${sender === 'ai' ? 'AI講師' : 'あなた'}:</strong> ${message}`;
            chatContainer.appendChild(messageDiv);
            chatContainer.scrollTop = chatContainer.scrollHeight;
        }

        function updateProgress(understanding, phase, progress) {
            document.getElementById('progress-fill').style.width = progress + '%';
            document.getElementById('progress-text').textContent = 
                `理解レベル: ${understanding}/10 | フェーズ: ${phase}`;
        }

        function getProgress() {
            fetch('/get_progress')
            .then(response => response.json())
            .then(data => {
                if (data.error) {
                    alert('エラー: ' + data.error);
                    return;
                }

                alert(`
学習進捗レポート:
• 理解レベル: ${data.understanding_level}/10
• 現在のフェーズ: ${data.current_phase}
• 完了目標: ${data.objectives_completed}/${data.total_objectives}
• 会話数: ${data.conversation_count}
                `);
            })
            .catch(error => {
                console.error('Error:', error);
                alert('進捗の取得に失敗しました');
            });
        }

        function handleKeyPress(event) {
            if (event.key === 'Enter') {
                submitResponse();
            }
        }
    </script>

# 必要なツールの確認とインストール
check_and_install() {
    local tool=$1
    local install_cmd=$2

    if ! command -v $tool &> /dev/null; then
        echo "📦 $tool をインストール中..."
        eval $install_cmd
    else
        echo "✅ $tool は既にインストール済み"
    fi
}

# 基本ツールのインストール

    # claude4_copilot_integration.py - 統合管理システム
import asyncio
import json
import subprocess
import time
from typing import Dict, List, Optional
from dataclasses import dataclass
from pathlib import Path

@dataclass 
class DevelopmentTask:
    id: str
    description: str
    priority: int  # 1-5
    estimated_time: int  # minutes
    assigned_ai: str  # 'claude', 'copilot', 'both'
    status: str  # 'pending', 'in_progress', 'completed'
    result: Optional[Dict] = None

class Claude4CopilotManager:
    def __init__(self, project_path: str):
        self.project_path = Path(project_path)
        self.tasks = []
        self.performance_metrics = {
            'tasks_completed': 0,
            'total_time_saved': 0,
            'code_quality_score': 0,
            'bugs_prevented': 0
        }

    async def add_task(self, description: str, priority: int = 3) -> str:
        """開発タスクを追加"""
        task_id = f"task_{int(time.time())}"

        # AI assignment logic
        assigned_ai = self._determine_ai_assignment(description)
        estimated_time = self._estimate_task_time(description)

        task = DevelopmentTask(
            id=task_id,
            description=description,
            priority=priority,
            estimated_time=estimated_time,
            assigned_ai=assigned_ai,
            status='pending'
        )

        self.tasks.append(task)
        return task_id

    def _determine_ai_assignment(self, description: str) -> str:
        """タスクの内容に基づいて最適なAIを選択"""
        description_lower = description.lower()

        # Code generation and refactoring -> Copilot
        if any(keyword in description_lower for keyword in 
               ['code', 'implement', 'refactor', 'function', 'class']):
            return 'copilot'

        # Analysis, architecture, documentation -> Claude
        elif any(keyword in description_lower for keyword in 
                ['analyze', 'architecture', 'design', 'documentation', 'explain']):
            return 'claude'

        # Complex tasks -> Both
        elif any(keyword in description_lower for keyword in 
                ['optimize', 'debug', 'performance', 'security']):
            return 'both'

        return 'claude'  # Default

    def _estimate_task_time(self, description: str) -> int:
        """タスクの推定時間を計算（分）"""
        word_count = len(description.split())

        if word_count < 5:
            return 15  # Simple task
        elif word_count < 15:
            return 30  # Medium task
        else:
            return 60  # Complex task

    async def execute_task(self, task_id: str) -> Dict:
        """タスクを実行"""
        task = next((t for t in self.tasks if t.id == task_id), None)
        if not task:
            return {"error": "Task not found"}

        task.status = 'in_progress'
        start_time = time.time()

        try:
            if task.assigned_ai == 'claude':
                result = await self._execute_claude_task(task)
            elif task.assigned_ai == 'copilot':
                result = await self._execute_copilot_task(task)
            else:  # both
                result = await self._execute_combined_task(task)

            task.result = result
            task.status = 'completed'

            # Update metrics
            execution_time = time.time() - start_time
            self.performance_metrics['tasks_completed'] += 1
            self.performance_metrics['total_time_saved'] += max(0, 
                task.estimated_time * 60 - execution_time)

            return result

        except Exception as e:
            task.status = 'failed'
            return {"error": str(e)}

    async def _execute_claude_task(self, task: DevelopmentTask) -> Dict:
        """Claude 4でタスクを実行"""
        cmd = [
            'claude', 'code', 
            '--prompt', task.description,
            '--project-path', str(self.project_path),
            '--format', 'json'
        ]

        result = subprocess.run(cmd, capture_output=True, text=True)

        if result.returncode == 0:
            try:
                return json.loads(result.stdout)
            except json.JSONDecodeError:
                return {"output": result.stdout, "type": "claude_response"}
        else:
            return {"error": result.stderr}

    async def _execute_copilot_task(self, task: DevelopmentTask) -> Dict:
        """GitHub Copilotでタスクを実行"""
        cmd = [
            'gh', 'copilot', 'suggest',
            task.description,
            '--format', 'json'
        ]

        result = subprocess.run(cmd, capture_output=True, text=True, 
                              cwd=self.project_path)

        if result.returncode == 0:
            return {"suggestion": result.stdout, "type": "copilot_suggestion"}
        else:
            return {"error": result.stderr}

    async def _execute_combined_task(self, task: DevelopmentTask) -> Dict:
        """Claude 4とCopilotを組み合わせてタスクを実行"""
        # Phase 1: Claude for analysis
        claude_result = await self._execute_claude_task(task)

        # Phase 2: Copilot for implementation
        implementation_task = DevelopmentTask(
            id=f"{task.id}_impl",
            description=f"Implement based on analysis: {claude_result.get('output', '')}",
            priority=task.priority,
            estimated_time=task.estimated_time // 2,
            assigned_ai='copilot',
            status='pending'
        )

        copilot_result = await self._execute_copilot_task(implementation_task)

        return {
            "analysis": claude_result,
            "implementation": copilot_result,
            "type": "combined_response"
        }

    def get_performance_report(self) -> Dict:
        """パフォーマンスレポートを生成"""
        completed_tasks = [t for t in self.tasks if t.status == 'completed']
        failed_tasks = [t for t in self.tasks if t.status == 'failed']

        return {
            "summary": {
                "total_tasks": len(self.tasks),
                "completed": len(completed_tasks),
                "failed": len(failed_tasks),
                "success_rate": len(completed_tasks) / len(self.tasks) if self.tasks else 0
            },
            "metrics": self.performance_metrics,
            "ai_usage": {
                "claude_only": len([t for t in completed_tasks if t.assigned_ai == 'claude']),
                "copilot_only": len([t for t in completed_tasks if t.assigned_ai == 'copilot']),
                "combined": len([t for t in completed_tasks if t.assigned_ai == 'both'])
            },
            "time_analysis": {
                "total_estimated": sum(t.estimated_time for t in self.tasks),
                "time_saved_hours": self.performance_metrics['total_time_saved'] / 3600
            }
        }

    async def continuous_development_mode(self, task_file: str):
        """継続的開発モード - ファイル監視してタスクを自動実行"""
        print("🚀 継続的開発モード開始")

        while True:
            try:
                if Path(task_file).exists():
                    with open(task_file, 'r') as f:
                        new_tasks = json.load(f)

                    for task_data in new_tasks:
                        task_id = await self.add_task(
                            task_data['description'],
                            task_data.get('priority', 3)
                        )
                        print(f"📝 新しいタスク追加: {task_id}")

                        # 即座に実行
                        result = await self.execute_task(task_id)
                        print(f"✅ タスク完了: {task_id}")
                        print(f"結果: {result.get('type', 'unknown')}")

                    # タスクファイルをクリア
                    Path(task_file).unlink()

                await asyncio.sleep(5)  # 5秒間隔で監視

            except KeyboardInterrupt:
                print("👋 継続的開発モード終了")
                break
            except Exception as e:
                print(f"❌ エラー: {e}")
                await asyncio.sleep(5)

    // mcp-integration.ts - MCPサーバー実装
import { Server } from '@modelcontextprotocol/sdk/server/index.js';
import { StdioServerTransport } from '@modelcontextprotocol/sdk/server/stdio.js';
import { 
    CallToolRequestSchema,
    ListToolsRequestSchema,
    Tool
} from '@modelcontextprotocol/sdk/types.js';

interface LearningContext {
    userId: string;
    currentTopic: string;
    understanding_level: number;
    session_history: Array<{role: string, content: string}>;
}

class AILearningMCPServer {
    private server: Server;
    private learningContexts: Map<string, LearningContext> = new Map();

    constructor() {
        this.server = new Server(
            { name: 'ai-learning-server', version: '1.0.0' },
            { capabilities: { tools: {} } }
        );

        this.setupHandlers();
    }

    private setupHandlers() {
        // ツール一覧の提供
        this.server.setRequestHandler(ListToolsRequestSchema, async () => ({
            tools: [
                {
                    name: 'create_learning_session',
                    description: 'Create a new Socratic learning session',
                    inputSchema: {
                        type: 'object',
                        properties: {
                            userId: { type: 'string' },
                            topic: { type: 'string' },
                            learningGoals: {
                                type: 'array',
                                items: { type: 'string' }
                            }
                        },
                        required: ['userId', 'topic']
                    }
                },
                {
                    name: 'generate_socratic_question',
                    description: 'Generate a Socratic method question',
                    inputSchema: {
                        type: 'object',
                        properties: {
                            userId: { type: 'string' },
                            studentResponse: { type: 'string' }
                        },
                        required: ['userId']
                    }
                },
                {
                    name: 'analyze_understanding',
                    description: 'Analyze student understanding level',
                    inputSchema: {
                        type: 'object',
                        properties: {
                            userId: { type: 'string' },
                            response: { type: 'string' }
                        },
                        required: ['userId', 'response']
                    }
                },
                {
                    name: 'get_learning_progress',
                    description: 'Get current learning progress',
                    inputSchema: {
                        type: 'object',
                        properties: {
                            userId: { type: 'string' }
                        },
                        required: ['userId']
                    }
                }
            ] as Tool[]
        }));

        // ツール呼び出しの処理
        this.server.setRequestHandler(CallToolRequestSchema, async (request) => {
            const { name, arguments: args } = request.params;

            switch (name) {
                case 'create_learning_session':
                    return await this.createLearningSession(args);

                case 'generate_socratic_question':
                    return await this.generateSocraticQuestion(args);

                case 'analyze_understanding':
                    return await this.analyzeUnderstanding(args);

                case 'get_learning_progress':
                    return await this.getLearningProgress(args);

                default:
                    throw new Error(`Unknown tool: ${name}`);
            }
        });
    }

    private async createLearningSession(args: any) {
        const { userId, topic, learningGoals = [] } = args;

        const context: LearningContext = {
            userId,
            currentTopic: topic,
            understanding_level: 5,
            session_history: [{
                role: 'system',
                content: `Learning session started for topic: ${topic}`
            }]
        };

        this.learningContexts.set(userId, context);

        return {
            content: [{
                type: 'text',
                text: `Learning session created for user ${userId} on topic: ${topic}`
            }]
        };
    }

    private async generateSocraticQuestion(args: any) {
        const { userId, studentResponse } = args;
        const context = this.learningContexts.get(userId);

        if (!context) {
            throw new Error(`No learning session found for user ${userId}`);
        }

        // Add student response to history
        if (studentResponse) {
            context.session_history.push({
                role: 'user',
                content: studentResponse
            });
        }

        // Generate Socratic question logic
        const question = await this.generateQuestion(context);

        context.session_history.push({
            role: 'assistant',
            content: question
        });

        return {
            content: [{
                type: 'text',
                text: question
            }]
        };
    }

    private async generateQuestion(context: LearningContext): Promise<string> {
        // Simplified question generation logic
        const questionTypes = [
            `What do you think is the most important aspect of ${context.currentTopic}?`,
            `Can you give me an example of how ${context.currentTopic} might be used in real life?`,
            `What questions do you have about ${context.currentTopic}?`,
            `How does ${context.currentTopic} relate to something you already know?`,
            `What would happen if we changed this aspect of ${context.currentTopic}?`
        ];

        const randomIndex = Math.floor(Math.random() * questionTypes.length);
        return questionTypes[randomIndex];
    }

    private async analyzeUnderstanding(args: any) {
        const { userId, response } = args;
        const context = this.learningContexts.get(userId);

        if (!context) {
            throw new Error(`No learning session found for user ${userId}`);
        }

        // Simplified understanding analysis
        const wordCount = response.split(' ').length;
        const hasExamples = response.toLowerCase().includes('example') || 
                           response.toLowerCase().includes('like');
        const hasQuestions = response.includes('?');

        let newLevel = context.understanding_level;

        if (wordCount > 20 && hasExamples) {
            newLevel = Math.min(10, context.understanding_level + 1);
        } else if (wordCount < 10 || hasQuestions) {
            newLevel = Math.max(1, context.understanding_level - 1);
        }

        context.understanding_level = newLevel;

        return {
            content: [{
                type: 'text',
                text: JSON.stringify({
                    understanding_level: newLevel,
                    analysis: {
                        word_count: wordCount,
                        has_examples: hasExamples,
                        has_questions: hasQuestions
                    }
                })
            }]
        };
    }

    private async getLearningProgress(args: any) {
        const { userId } = args;
        const context = this.learningContexts.get(userId);

        if (!context) {
            throw new Error(`No learning session found for user ${userId}`);
        }

        const progress = {
            userId: context.userId,
            topic: context.currentTopic,
            understanding_level: context.understanding_level,
            conversation_turns: context.session_history.length,
            progress_percentage: Math.min(100, (context.understanding_level / 10) * 100)
        };

        return {
            content: [{
                type: 'text',
                text: JSON.stringify(progress, null, 2)
            }]
        };
    }

    async start() {
        const transport = new StdioServerTransport();
        await this.server.connect(transport);
        console.error('AI Learning MCP Server started');
    }
}
